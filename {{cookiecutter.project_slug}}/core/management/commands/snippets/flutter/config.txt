///
/// [getRemoteConfigExample]
///
/// Método de exemplo para recuperar uma chave do Firebase Remote Config, 
/// 
/// [Params]:
///  String nomeChaveFirebaseConfig - Nome da chave a ser recuperada
///
/// [Return]:
///  String - Valor da chave buscada ou ''
///
Future<String> getRemoteConfigExample(String nomeChaveFirebaseConfig) async {
  try {
    if (releaseVersion == false) {
      // Caso esteja em desenvolvimento e desejar retornar um valor fixo basta alterar a linha abaixo.
      return '';
    }
    final String? config = await ApplicationConfig.getFirebaseRemoteConfig(nomeChaveFirebaseConfig);
    if (config != null) {
      return config;
    }
    return '';
  } catch (error, stackTrace) {
    AgtecCoreLogger().erro('Erro ao chamar o getRemoteConfigExample', error, stackTrace);
    return '';
  }
}
